월별 & 성별 (5월 ~ 9월)

for (i in 5:9)
{
    man <- subset(df, 매출월 == i & 성별 == "남")
    man.mean <- mean(man$구매건수)

    woman <- subset(df, 매출월 == i & 성별 == "여")
    woman.mean <- mean(woman$구매건수)

    png(filename = sprintf("%d%s", i, ".png"))
    barplot(c(man.mean, woman.mean), 
        names.arg = c("남", "여"),
        main = paste(i, "월 남녀 구매건수 평균"),
        col = c("red", "blue"),
        density = 30,
        legend = c("남", "여"))
    dev.off()
}
